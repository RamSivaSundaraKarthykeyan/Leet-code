/**
 * @param {number[]} nums
 * @return {number}
 */
var largestPerimeter = function(nums) {
     nums = nums.sort((a, b) => a - b);
    let res = 0;
    for(let i = 0; i < nums.length - 2; i++){
        let first = nums[i];
        let second = nums[i+1];
        let third = nums[i+2];

        if(first + second > third && first + third > second && second + third > first){
            res = Math.max(res, first + second + third);
        }
    }
    return res;
};
